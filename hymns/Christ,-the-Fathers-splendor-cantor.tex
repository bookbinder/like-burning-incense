\hymn

\psalmsubtitle{Christ, the Fathers Splendor}

\lilypondfile{/home/ryan/scores/like-burning-incense/hymns/Christ,-the-Fathers-splendor-cantor.ly}

\begin{underhymnverse}
\flagverse{\footnotesize\textbf{5.}}
This poor home surpasses\\
ev’ry court and palace,\\
harboring salvation\\
for the human fam’ly.

\flagverse{\footnotesize\textbf{6.}}
Jesus, Mary, Joseph,\\
grant us and our fam’lies\\
joy and gifts most holy\\
from your humble dwelling.

\flagverse{\footnotesize\textbf{7.}}
Praise to you, Christ Jesus,\\
for to us you offer\\
hope that through your parents,\\
we come home to heaven. Amen.
\end{underhymnverse}
