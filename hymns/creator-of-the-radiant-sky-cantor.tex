\hymn

\psalmsubtitle{Creator of the Radiant Sky}

\lilypondfile{/home/ryan/scores/like-burning-incense/hymns/creator-of-the-radiant-sky-cantor.ly}

\begin{underhymnverse}
\flagverse{\footnotesize\textbf{5.}}
But, as the sun traverses day,\\
may faith be ardent, firm, and deep,\\
may hope incite to promised good,\\
may charity wed us to Christ.

\flagverse{\footnotesize\textbf{6.}}
Most loving Father, hear our prayer,\\
and you, O Christ coequal Son,\\
who with the Spirit Paraclete\\
now reign for all eternity. Amen.
\end{underhymnverse}
