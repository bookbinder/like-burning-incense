\hymn

\psalmsubtitle{Receive, O Mary, Virgin Pure}

\lilypondfile[line-width=3.73in]{/home/ryan/scores/like-burning-incense/hymns/receive,-o-mary,-virgin-pure-cantor.ly}

\begin{underhymnverse}
\flagverse{\footnotesize\textbf{5.}}
Rejoice together, angel hosts!\\
Let all the peoples sing for joy.\\
The Most High comes, a humble child,\\
to save and ransom what was lost.

\flagverse{\footnotesize\textbf{6.}}
To you, O Christ, most loving King,\\
and to the Father, glory be,\\
one with the Spirit Paraclete,\\
from age to age for evermore. Amen.
\end{underhymnverse}

Text: \emph{Verbum salutis omnium}, ca. 12th c., © 2023 ICEL
