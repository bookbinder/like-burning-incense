\psalmsubtitle{Eternal Glory of the Sky}

\lilypondfile[line-width=3.73in]{/home/ryan/scores/like-burning-incense/hymns/eternal-glory-of-the-sky-cantor.ly}

\setlength{\vleftmargin}{2em}
\begin{verse}
\flagverse{\footnotesize\textbf{5.}}
The Faith that first must be possess'd\\
Root deep within our inmost breast:\\
And joyous Hope in second place,\\
Then Charity, thy greatest grace.

\flagverse{\footnotesize\textbf{6.}}
All laud to God the Father be;\\
All praise, eternal Son, to thee:\\
All glory, as is ever meet,\\
To God the Holy Paraclete. Amen.
\end{verse}
