\psalmsubtitle{Behold the Fiery Sun Arise!}

\lilypondfile[line-width=3.73in]{/home/ryan/scores/like-burning-incense/hymns/behold-the-fiery-sun-arise-cantor.ly}

\setlength{\vleftmargin}{2em}
\begin{verse}
\flagverse{\footnotesize\textbf{5.}}
The One who sees stands by on high\\
and watches us through all our days,\\
discerning ev’ry act and deed\\
from early dawn to eventide.

\flagverse{\footnotesize\textbf{6.}}
To God the Father glory be,\\
all honor to his only Son,\\
one with the Spirit Paraclete,\\
from age to age for evermore. Amen.
\end{verse}
